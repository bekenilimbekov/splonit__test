git init — Инициализация нового репозитория.
git add . — Добавление всех измененных файлов в индекс.
git commit -m "Сообщение" — Сохранение изменений с комментарием.
git status — Просмотр статуса репозитория (что изменено, что добавлено).
git log — Просмотр истории коммитов.
git remote add origin <url> — Добавление удаленного репозитория.
git push origin main — Отправка изменений на удаленный репозиторий (ветка main).
git pull origin main — Загрузка последних изменений с удаленного репозитория.
git checkout <ветка> — Переключение на другую ветку.
git branch <имя-ветки> — Создание новой ветки.
git clone <url> — Клонирование репозитория с удалённого сервера на твой компьютер.
git merge <ветка> — Слияние указанной ветки с текущей.
git branch — Список всех веток в репозитории.
git branch -d <имя-ветки> — Удаление локальной ветки.
git reset — Отмена изменений, возвращение к последнему коммиту.
git stash — Временное сохранение изменений (сохранить изменения и вернуться к последнему коммиту).
git stash pop — Восстановление сохранённых изменений после использования git stash.
git diff — Просмотр различий между текущими изменениями и последним коммитом.
git fetch — Загрузка изменений с удалённого репозитория, но без слияния.
git rebase <ветка> — Перемещение текущей ветки на новую основу (вместо слияния).
